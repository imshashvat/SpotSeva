// lib/presentation/bloc/spot/spot_event.dart
abstract class SpotEvent {}

class LoadNearbySpots extends SpotEvent {
  final double lat, lng;
  final double radiusMeters;
  LoadNearbySpots(this.lat, this.lng, {this.radiusMeters = 500});
}

class AdoptSpot   extends SpotEvent { final String spotId; AdoptSpot(this.spotId); }
class ReleaseSpot extends SpotEvent { final String spotId; ReleaseSpot(this.spotId); }
class CheckInSpot extends SpotEvent {
  final String spotId;
  final double lat, lng;
  CheckInSpot(this.spotId, this.lat, this.lng);
}
// lib/presentation/bloc/spot/spot_state.dart
abstract class SpotState {}

class SpotInitial  extends SpotState {}
class SpotLoading  extends SpotState {}
class SpotError    extends SpotState { final String message; SpotError(this.message); }

class SpotsLoaded extends SpotState {
  final List<Spot> spots;
  SpotsLoaded(this.spots);
}

class AdoptionSuccess extends SpotState {
  final int pointsEarned;
  AdoptionSuccess(this.pointsEarned);
}

class CheckInSuccess extends SpotState {
  final int pointsEarned;
  final int currentStreak;
  CheckInSuccess(this.pointsEarned, this.currentStreak);
}
// lib/presentation/bloc/spot/spot_bloc.dart
class SpotBloc extends Bloc<SpotEvent, SpotState> {
  final GetNearbySpots  _getNearbySpots;
  final AdoptSpotUseCase _adoptSpot;
  final CheckInUseCase   _checkIn;

  SpotBloc({
    required GetNearbySpots  getNearbySpots,
    required AdoptSpotUseCase adoptSpot,
    required CheckInUseCase   checkIn,
  }) : _getNearbySpots = getNearbySpots,
       _adoptSpot      = adoptSpot,
       _checkIn        = checkIn,
       super(SpotInitial()) {
    on<LoadNearbySpots>(_onLoad);
    on<AdoptSpot>(_onAdopt);
    on<CheckInSpot>(_onCheckIn);
  }

  Future<void> _onLoad(LoadNearbySpots event, Emitter emit) async {
    emit(SpotLoading());
    final result = await _getNearbySpots(
      NearbyParams(lat: event.lat, lng: event.lng, radius: event.radiusMeters)
    );
    result.fold(
      (failure) => emit(SpotError(failure.message)),
      (spots)   => emit(SpotsLoaded(spots)),
    );
  }

  Future<void> _onAdopt(AdoptSpot event, Emitter emit) async {
    emit(SpotLoading());
    final result = await _adoptSpot(AdoptParams(spotId: event.spotId));
    result.fold(
      (f)    => emit(SpotError(f.message)),
      (resp) => emit(AdoptionSuccess(resp.pointsEarned)),
    );
  }

  Future<void> _onCheckIn(CheckInSpot event, Emitter emit) async {
    emit(SpotLoading());
    final result = await _checkIn(
      CheckInParams(spotId: event.spotId, lat: event.lat, lng: event.lng)
    );
    result.fold(
      (f)    => emit(SpotError(f.message)),
      (resp) => emit(CheckInSuccess(resp.pointsEarned, resp.streak)),
    );
  }
}

ReportBloc

// lib/presentation/bloc/report/report_bloc.dart
class ReportBloc extends Bloc<ReportEvent, ReportState> {
  final SubmitReportUseCase _submitReport;
  final FirebaseStorage    _storage;

  ReportBloc({required SubmitReportUseCase submitReport, required FirebaseStorage storage})
    : _submitReport = submitReport, _storage = storage, super(ReportInitial()) {
    on<SubmitReport>(_onSubmit);
    on<UploadPhoto>(_onUpload);
  }

  Future<void> _onUpload(UploadPhoto event, Emitter emit) async {
    emit(ReportUploading(progress: 0));
    final ref = _storage.ref('reports/${DateTime.now().millisecondsSinceEpoch}.jpg');
    final task = ref.putFile(event.imageFile);
    task.snapshotEvents.listen((snap) {
      final prog = snap.bytesTransferred / snap.totalBytes;
      emit(ReportUploading(progress: prog));
    });
    final url = await (await task).ref.getDownloadURL();
    emit(PhotoUploaded(url: url));
  }

  Future<void> _onSubmit(SubmitReport event, Emitter emit) async {
    emit(ReportSubmitting());
    final result = await _submitReport(ReportParams(
      spotId: event.spotId, photoUrls: event.photoUrls,
      issueType: event.issueType, description: event.description,
    ));
    result.fold(
      (f)    => emit(ReportError(f.message)),
      (resp) => emit(ReportSuccess(reportId: resp.reportId, pointsEarned: resp.pointsEarned, aiLabel: resp.aiLabel)),
    );
  }
}

class MapScreen extends StatefulWidget {
  @override
  State<MapScreen> createState() => _MapScreenState();
}

class _MapScreenState extends State<MapScreen> {
  final Completer<GoogleMapController> _ctrl = Completer();
  Set<Marker> _markers = {};

  @override
  void initState() {
    super.initState();
    _loadLocation();
  }

  Future<void> _loadLocation() async {
    final pos = await Geolocator.getCurrentPosition();
    context.read<SpotBloc>().add(LoadNearbySpots(pos.latitude, pos.longitude));
  }

  Set<Marker> _buildMarkers(List<Spot> spots) {
    return spots.map((spot) => Marker(
      markerId: MarkerId(spot.id),
      position: LatLng(spot.geopoint.latitude, spot.geopoint.longitude),
      icon: _iconForStatus(spot.status),
      onTap: () => _showSpotSheet(spot),
    )).toSet();
  }

  BitmapDescriptor _iconForStatus(SpotStatus s) => switch (s) {
    SpotStatus.clean    => BitmapDescriptor.defaultMarkerWithHue(BitmapDescriptor.hueGreen),
    SpotStatus.issue    => BitmapDescriptor.defaultMarkerWithHue(BitmapDescriptor.hueOrange),
    SpotStatus.critical => BitmapDescriptor.defaultMarkerWithHue(BitmapDescriptor.hueRed),
    SpotStatus.adopted  => BitmapDescriptor.defaultMarkerWithHue(BitmapDescriptor.hueViolet),
  };

  void _showSpotSheet(Spot spot) => showModalBottomSheet(
    context: context,
    builder: (_) => SpotDetailSheet(spot: spot),
  );

  @override
  Widget build(BuildContext context) {
    return BlocListener<SpotBloc, SpotState>(
      listener: (ctx, state) {
        if (state is SpotsLoaded) setState(() => _markers = _buildMarkers(state.spots));
        if (state is CheckInSuccess) _showCheckInToast(ctx, state);
      },
      child: Stack(children: [
        GoogleMap(
          onMapCreated: _ctrl.complete,
          initialCameraPosition: const CameraPosition(
            target: LatLng(28.4744, 77.5040), // Greater Noida
            zoom: 15,
          ),
          markers: _markers,
          myLocationEnabled: true,
          myLocationButtonEnabled: false,
        ),
        const _NearbySpotsFAB(),
      ]),
    );
  }
}

SpotRepositoryImpl — Firestore + Geohash Queries

// lib/data/repositories/spot_repository_impl.dart
class SpotRepositoryImpl implements SpotRepository {
  final FirebaseFirestore _db;
  const SpotRepositoryImpl(this._db);

  @override
  Future<Either<Failure, List<Spot>>> getNearbySpots({
    required double lat, required double lng, double radius = 500
  }) async {
    try {
      // Geohash bounding box — queries without extra index
      final bounds = Geoflutter.boundingBoxCoordinates(
        center: GeoPoint(lat, lng), radiusInKm: radius / 1000
      );
      final snap = await _db.collection('spots')
        .where('geohash', isGreaterThanOrEqualTo: bounds.swHash)
        .where('geohash', isLessThanOrEqualTo: bounds.neHash)
        .get();
      return Right(snap.docs.map(SpotModel.fromDoc).toList());
    } on FirebaseException catch (e) {
      return Left(ServerFailure(e.message ?? 'Firestore error'));
    }
  }

  @override
  Future<Either<Failure, AdoptResponse>> adoptSpot(String spotId) async {
    try {
      final callable = FirebaseFunctions.instance.httpsCallable('adoptSpot');
      final result = await callable.call({'spotId': spotId});
      return Right(AdoptResponse.fromMap(result.data as Map));
    } on FirebaseFunctionsException catch (e) {
      return Left(ServerFailure(e.message ?? 'Function error'));
    }
  }
}

class SpotModel extends Spot {
  const SpotModel({required super.id, /* ... */});

  factory SpotModel.fromDoc(DocumentSnapshot doc) {
    final d = doc.data()! as Map<String, dynamic>;
    return SpotModel(
      id:             doc.id,
      name:           d['name'] as String,
      adopterId:      d['adopterId'] as String? ?? '',
      geohash:        d['geohash'] as String,
      geopoint:       d['geopoint'] as GeoPoint,
      status:         SpotStatus.values.byName(d['status'] as String),
      category:       d['category'] as String,
      checkinsCount:  (d['checkinsCount'] as int?) ?? 0,
      lastCheckin:    (d['lastCheckin'] as Timestamp).toDate(),
    );
  }

  Map<String, dynamic> toMap() => {
    'name':          name,
    'adopterId':     adopterId,
    'geohash':       geohash,
    'geopoint':      geopoint,
    'status':        status.name,
    'category':      category,
    'checkinsCount': checkinsCount,
    'lastCheckin':   Timestamp.fromDate(lastCheckin),
  };
}

onCheckIn — GPS Validation + Points + Streak

// functions/src/checkin.ts
import * as functions from 'firebase-functions';
import * as admin    from 'firebase-admin';
import { distanceBetween } from 'geofire-common';

const db = admin.firestore();
const CHECKIN_RADIUS_M = 100;
const CHECKIN_POINTS   = 10;
const STREAK_7_BONUS   = 50;
const STREAK_30_BONUS  = 200;

export const onCheckIn = functions.firestore
  .document('checkins/{checkinId}')
  .onCreate(async (snap, ctx) => {
    const { userId, spotId, geopoint } = snap.data();

    // 1. Fetch spot and validate proximity
    const spotDoc = await db.collection('spots').doc(spotId).get();
    if (!spotDoc.exists) throw new Error('Spot not found');
    const spot = spotDoc.data()!;
    const dist = distanceBetween(
      [geopoint.latitude, geopoint.longitude],
      [spot.geopoint.latitude, spot.geopoint.longitude]
    ) * 1000; // km → m
    if (dist > CHECKIN_RADIUS_M) {
      await snap.ref.update({ valid: false, error: 'Too far from spot' });
      return;
    }

    // 2. Run atomic transaction: update wallet + streak + spot
    await db.runTransaction(async (t) => {
      const walletRef = db.collection('wallets').doc(userId);
      const wallet    = (await t.get(walletRef)).data() ?? { balance: 0, streak: 0, lastCheckin: null };

      // Streak logic
      const now      = new Date();
      const lastDate = wallet.lastCheckin?.toDate() ?? new Date(0);
      const daysDiff = (now.getTime() - lastDate.getTime()) / (1000 * 60 * 60 * 24);
      const newStreak = daysDiff <= 1.5 ? wallet.streak + 1 : 1;
      const bonus     = newStreak === 7 ? STREAK_7_BONUS : newStreak === 30 ? STREAK_30_BONUS : 0;
      const total     = CHECKIN_POINTS + bonus;

      t.update(walletRef, {
        balance:       admin.firestore.FieldValue.increment(total),
        lifetimeEarned: admin.firestore.FieldValue.increment(total),
        streak:         newStreak,
        lastCheckin:    admin.firestore.Timestamp.now(),
      });

      t.update(db.collection('spots').doc(spotId), {
        checkinsCount: admin.firestore.FieldValue.increment(1),
        lastCheckin:   admin.firestore.Timestamp.now(),
      });

      t.update(snap.ref, { valid: true, pointsEarned: total, newStreak });
    });
  });

import { GoogleGenerativeAI } from '@google/generative-ai';

const genAI  = new GoogleGenerativeAI(process.env.GEMINI_API_KEY!);

export const onIssueReport = functions.firestore
  .document('reports/{reportId}')
  .onCreate(async (snap) => {
    const { reporterId, spotId, photoUrls, issueType } = snap.data();

    // 1. Classify with Gemini Flash (vision)
    let aiLabel = issueType, severity = 'medium';
    try {
      const result = await model.generateContent([
        {
          inlineData: {
            mimeType: 'image/jpeg',
            data: await fetchImageBase64(photoUrls[0]),
          }
        },
        `Classify this civic issue image. Respond with JSON only:
         {"label": "Broken bench|Fallen tree|Garbage dump|Pothole|Graffiti|Other",
          "severity": "low|medium|high",
          "confidence": 0.0-1.0}`
      ]);
      const parsed = JSON.parse(result.response.text());
      aiLabel  = parsed.label;
      severity = parsed.severity;
    } catch (e) {
      console.error('Gemini classification failed', e);
    }

    // 2. Atomic: award points + update spot status + write AI label
    await db.runTransaction(async (t) => {
      const walletRef = db.collection('wallets').doc(reporterId);
      t.update(walletRef, {
        balance:        admin.firestore.FieldValue.increment(REPORT_POINTS),
        lifetimeEarned: admin.firestore.FieldValue.increment(REPORT_POINTS),
      });
      t.update(db.collection('spots').doc(spotId), {
        status: severity === 'high' ? 'critical' : 'issue',
      });
      t.update(snap.ref, { aiLabel, severity, pointsEarned: REPORT_POINTS, status: 'open' });
    });

    // 3. Alert municipal FCM topic
    await admin.messaging().send({
      topic:        'municipal_alerts',
      notification: { title: `New ${severity} issue`, body: aiLabel },
      data:         { reportId: snap.id, spotId, severity },
    });
  });

onCouponRedeem — Atomic Transaction

// functions/src/rewards.ts
export const redeemCoupon = functions.https.onCall(async (data, context) => {
  if (!context.auth) throw new functions.https.HttpsError('unauthenticated', 'Login required');
  const { rewardId } = data;
  const userId = context.auth.uid;

  return db.runTransaction(async (t) => {
    const rewardRef = db.collection('rewards').doc(rewardId);
    const walletRef = db.collection('wallets').doc(userId);
    const [rewardDoc, walletDoc] = await Promise.all([t.get(rewardRef), t.get(walletRef)]);

    const reward = rewardDoc.data()!;
    const wallet = walletDoc.data()!;

    if (wallet.balance < reward.pointsCost)
      throw new functions.https.HttpsError('failed-precondition', 'Insufficient points');
    if (reward.redeemedQty >= reward.totalQty)
      throw new functions.https.HttpsError('failed-precondition', 'Coupon sold out');

    const couponCode = generateCode(); // crypto random 8-char
    t.update(walletRef, { balance: admin.firestore.FieldValue.increment(-reward.pointsCost) });
    t.update(rewardRef, { redeemedQty: admin.firestore.FieldValue.increment(1) });
    t.set(db.collection('redemptions').doc(), {
      userId, rewardId, couponCode, redeemedAt: admin.firestore.Timestamp.now(), used: false
    });
    return { couponCode, newBalance: wallet.balance - reward.pointsCost };
  });
});

    const doc = await checkinRef.get();
    expect(doc.data()?.valid).toBe(false);
  });

Flutter App Config

  • Add google-services.json (Android) + GoogleService-Info.plist (iOS)
  • Set Google Maps API key in AndroidManifest.xml + Info.plist
  • Configure .env with GEMINI_API_KEY (for direct calls)
  • Set minimum SDK: Android 21 (5.0+), iOS 13+
  • Configure location permissions (foreground + background)
  • Add camera + storage permissions in manifests
  • Set up release keystore + sign APK
  • Test FCM on real device (emulator won't receive push)
  • Test GPS proximity with real coordinates in field
  • Test offline Firestore cache (airplane mode scenario)

Firestore Indexes Required

  • spots: (geohash ASC, status ASC) — nearby + filtered
  • reports: (spotId ASC, status ASC, createdAt DESC) — spot history
  • reports: (status ASC, severity DESC, createdAt DESC) — municipal queue
  • checkins: (userId ASC, spotId ASC, timestamp DESC) — per-user history
  • leaderboard: (weekPoints DESC) — weekly ranking
  • rewards: (category ASC, pointsCost ASC) — store browse

Security Hardening

  • Firestore rules: wallets write = false (Functions only)
  • Firestore rules: reports status update = municipal role only
  • Firebase App Check enforced on production
  • GPS coordinates validated server-side (never trust client distance)
  • Rate limit check-ins: max 2/day/spot per user (Cloud Function)
  • Coupon redeem: atomic transaction prevents double-spend
  • Never expose Gemini API key in Flutter app — route through Functions
  • Validate all photo URLs match Firebase Storage domain before processing
